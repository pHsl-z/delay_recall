功能说明

  • ✅ 已实现：保存日托天数到user_settings表
  • ✅ 已实现：在添加新学生时自动加载记忆值

预期行为

  1. 用户首次添加日托学生时，输入自定义天数（如15天）
  2. 点击保存时，系统会将该天数保存到user_settings表的last_care_days字段
  3. 下次用户添加新的日托学生时，系统会自动加载上次保存的天数
  4. 在自定义天数输入框上方会显示"记忆值：X天"的蓝色标签

功能验证代码

以下是实现的核心代码片段：

// 获取用户日托天数记忆
async function getUserCareDaysMemory() {
    try {
        const { data: { user } } = await supabase.auth.getUser();
        if (!user) return null;
        
        const { data, error } = await supabase
            .from('user_settings')
            .select('last_care_days')
            .eq('user_id', user.id)
            .single();
            
        if (error) {
            console.log('未找到用户设置，返回null');
            return null;
        }
        
        return data?.last_care_days || null;
    } catch (error) {
        console.error('获取日托天数记忆失败:', error);
        return null;
    }
}

// 保存用户日托天数记忆
async function saveUserCareDaysMemory(careDays) {
    try {
        const { data: { user } } = await supabase.auth.getUser();
        if (!user) return;
        
        const { error } = await supabase
            .from('user_settings')
            .upsert({
                user_id: user.id,
                last_care_days: careDays,
                updated_at: new Date().toISOString()
            });
            
        if (error) {
            console.error('保存日托天数记忆失败:', error);
        } else {
            console.log('日托天数记忆已保存:', careDays);
        }
    } catch (error) {
        console.error('保存日托天数记忆失败:', error);
    }
}
